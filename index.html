فك تشفير Base64

Base64 هو نظام ترميح (encoding) وليس تشفيراً (encryption) حقيقياً، ولكنه يستخدم بشكل شائع في العديد من التطبيقات.

// دالة فك ترميح Base64 في JavaScript
function decodeBase64(str) {
  return atob(str);
}

فك تشفير XOR

XOR هو عملية ثنائية بسيطة تستخدم في التشفير. تتطلب معرفة المفتاح الصحيح لفك التشفير.

// دالة فك تشفير XOR في JavaScript
function decryptXOR(str, key) {
  let result = '';
  for (let i = 0; i < str.length; i++) {
    result += String.fromCharCode(str.charCodeAt(i) ^ key.charCodeAt(i % key.length));
  }
  return result;
}

فك تشفير AES

AES (معيار التشفير المتقدم) هو معيار تشفير قوي يستخدم على نطاق واسع.

// استخدام مكتبة CryptoJS لفك تشفير AES
function decryptAES(encrypted, key) {
  const bytes = CryptoJS.AES.decrypt(encrypted, key);
  return bytes.toString(CryptoJS.enc.Utf8);
}

فك تشفير شيفرة قيصر

شيفرة قيصر هي واحدة من أقدم تقنيات التشفير، حيث يتم إزاحة الحروف بعدد معين من المواضع.

// دالة فك تشفير شيفرة قيصر في JavaScript
function decryptCaesar(str, shift) {
  return str.replace(/[a-z]/gi, c => {
    const base = c < 'a' ? 65 : 97;
    return String.fromCharCode((c.charCodeAt(0) - base - shift + 26) % 26 + base);
  });
}
